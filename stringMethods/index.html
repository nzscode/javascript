String Methods
let str1 = "The boy who lived";
let str2 = "Now come to die!";
charAt( )
str1.charAt(6)  //y
returns character at index 6.
concat( )
str1.concat(str2)  //The boy who livedNow come to die!
returns str1 appended to str2.
length( )
str1.length(6)  //17
returns number of indexes.
indexOf( )
str1.indexOf("b")  //4
returns index of "b".
lastIndexOf( )
str1.lastIndexOf("e")  //15
returns last index of "e".
match( )
str1.match("who")  //
['who, index: 8, input: 'The boy who lived', groups: undefined]
normally matches against RegExp(regexp).
replace( )
str1.replace("boy", "girl")  //The girl who lived
returns string where "boy" is replaced with "girl".
search( )
str1.search("lived")  //12
normally matches against RegExp(regexp).
repeat( )
str1.repeat(2)  //The boy who livedThe boy who lived
returns the string repeated by the number specified.
trimEnd( )
str1.trimEnd()
returns string with extra whitespace removed from end.
trimRight( )
str1.trimRight()
returns string with extra whitespace removed from right of string.
slice( )
str1.slice(4, 10)  //boy who
returns string from index 4 upto but excluding index 10.
split( )
str1.split(" ")  //['The', 'boy', 'who', 'lived']
returns an array using " "(space) as a delimiter.
substring( )
str1.substring(5, 9)  //oy w
returns string from index 5 upto but excluding index 9.
toLowerCase( )
str1.toLowerCase(str1)  //the boy who lived
returns complete string in lower case.
toUpperCase( )
str1.toUpperCase(str1)  //THE BOY WHO LIVED
returns complete string in upper case.
includes( )
str1.includes("who")  //true
returns boolean if keyword is included inside string.
padEnd( )
str1.padEnd(252, "!")  //The boy who lived!!!!!!!!
returns string with "!" at end until total string length of 25 is reached.
padStart( )
str1.padStart(25, "!)  //!!!!!!!!The boy who lived
returns string with "!" at the start until total string length of 25 is reached.
startsWith( )
str1.startsWith(The)  //true
returns boolean if string starts with string.
endsWith( )
str1.endsWith("ed")  //true
returns boolean if string ends with string.
trim( )
str1.trim()  //
returns string with extra whitespace removed from right and end of string.
