Accessing objects:
Use bracket notation, only if key is known at runtime.
let myKey1 = myObj['key1'];
OR
let myKey1 = myObj.key1;
let myCarColor = Cars['color']
OR
let myCarColor = Cars.color;

console.log(myCarColor);
{color: 'pink'}
Assigning values to objects:
myObj['key1'] = 'Value1';
OR
myObj.key1 = 'value1';
Cars['color'] = 'pink';
OR
Cars.color = 'pink';

console.log(Cars);
{color: 'pink'}


Deleting key:values from objects:

Getters
Allows future access to specific Keys.

return this.key1;
}
};

console.log(myObj.getterName);
let Student = {
fName: "Jane",
lName: "Doe",

get getFirstName(){

}
};

Making the new defined property attribute over-writable:
Changing the writable attribute. (default = false)
Can this privately declared value be over-written as a property of myObj?
let myObj = {
key1: value1
}

Object.defineProperty(myObj, newKeyName,{
value: "value2",
writable: true
})


myObj.newKeyName = "VALUE2";

console.log(myObj);
let School = {
name: "Da Vinci PS"
};

Object.defineProperty(Student, 'district',{
value: "Durham Region",
writable: true
});

School.district = "Toronto Region";

console.log(School);
{name: 'Da Vinci PS', district: 'Toronto Region'}

Changing the enumerable attribute. (default = false)
Can this privately declared key/value be iterated as if part of myObj?

}

};

age
