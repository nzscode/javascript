Converting a Function Expression to an Arrow Function:
//Regular Function Expression
let greet = function (name) {
    console.log (`Hello ${name}`);
}

greet();


//Converted to an Arrow Function
let greetArrow = ((name) => {
    console.log (`Hello ${name}`);
})

greetArrow();

Output
Hello undefined
Hello undefined
Functions
//Regular Function Expression
let greetExpression = function (name) {
    console.log (`Function Expression => Hello ${name}`);
}

greetExpression();


//Regular Function Declaration
function greetDeclaration (name) {
    console.log (`Function Declaration => Hello ${name}`);
}

greetDeclaration();

Output
Function Expression => Hello undefined
Function Declaration => Hello undefined
Async
//Async Example
 console.log (1);
 console.log (2);
 setTimeout (() => {
    console.log (3);
 }, 2000);
 console.log (4);
 console.log (5);

Output
1
2
4
5
3
In the above program, the setTimeout function runs after 2000 milliseconds (2 seconds),
thus displaying the "3" after the rest of the code has run.
Callback Functions
//Main Function
let mainFunction = ((callback) => {
    console.log ('I am the main function, The callback is an argument of mine');
    callback();
});


//Second Function that is the callback
let secondFunction = (() => {
    console.log ('I am the second function, I am the callback function for main.');
});

mainFunction(secondFunction);

Output
I am the main function, The callback is an argument of mine
I am the second function, I am the callback function for main.




Another Example of a callback function
//the pizza is assembled only once the dough is risen, in the meantime the other ingredients can be prepped

let makingPizza = (() => {
    console.log ('Putting olives on Pizza');
    console.log ('Putting mushrooms on Pizza');
});


let doughPrep=((callback) => {
    console.log ('Mix yeast with water');
    console.log ('Mix flour with yeast);
    console.log ('Mix in salt');
    console.log ('let rise 2 hours');

//once dough is risen use it to make pizza
    callback();
});

doughPrep(makingPizza);
Output
Mix yeast with water
Mix flour with yeast
Mix in salt
let rise 2 hours
Putting olives on pizza
Putting mushrooms on pizza
