Concat ()

  • returns arr1 concatenated to arr2.
Code
let arr1 = [31, 74, 11, 8, 25, 11];
let arr2 = ['p', 'a', 'g', 'u', 'a'];

arr1.concat();

//[31, 74, 11, 8, 25, 11, 'p', 'a', 'g', 'u', 'a']


join( )

  • returns string by using delimiter between elements of the array.
Code
let arr1 = [31, 74, 11, 8, 25, 11];
let arr2 = ['p', 'a', 'g', 'u', 'a'];

arr1.join("!")
//31!74!11!8!25!11

sort( )

  • returns array sorted alphabetically.
Code
let arr1 = [31, 74, 11, 8, 25, 11];
let arr2 = ['p', 'a', 'g', 'u', 'a'];

arr1.sort()
// [11, 11, 25, 31, 74, 8] because 11 > 8

arr2.sort()
// ['a', 'a', 'g', 'p', 'u']


To Sort Numerically

Code

let arr2 = ['p', 'a', 'g', 'u', 'a'];

arr1.sort(function (a, b) {
    return a - b;
});

// [8, 11, 11, 25, 31, 74]
